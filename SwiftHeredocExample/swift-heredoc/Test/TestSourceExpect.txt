import Foundation

class Foo {

    // -- 標準パターン

    /*
     <<[EOL]
     Banana
     Apple
     Orange
     */
    let string1 = "Banana\nApple\nOrange"

    // -- 文字列のエスケープ

    /*
     <<[EOL]
     "" is a String literal.
     \r\n is line feed character.
     */
    let string2 = "\"\" is a String literal.\n\\r\\n is line feed character."

    // -- JSON文字列

    /*
     <<[EOL]
     {
         "resultCount": 1,
         "users": [{
             "id": "fooid",
             "name": "barname"
         }]
     }
     */
    let string3 = "{\n    \"resultCount\": 1,\n    \"users\": [{\n        \"id\": \"fooid\",\n        \"name\": \"barname\"\n    }]\n}"
}
