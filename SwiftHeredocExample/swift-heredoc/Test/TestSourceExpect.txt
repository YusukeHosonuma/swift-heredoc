import Foundation

class Foo {

    /*
     <<[EOL]
     Banana
     Apple
     Orange

    // -- 文字列のエスケープ

    /*
     <<[EOL]
     "Banana"

     */

    // -- JSON文字列

    /*
     <<[EOL]
     {

         "users": [{
             "id": "fooid",
             "name": "barname"
         }]
     }
     */
    let string3 = "{\n    \"resultCount\": 1,\n    \"users\": [{\n        \"id\": \"fooid\",\n        \"name\": \"barname\"\n    }]\n}"
